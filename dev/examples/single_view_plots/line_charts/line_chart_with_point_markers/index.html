Line chart with point markers

using Deneb

data = Data(url="https://vega.github.io/vega-datasets/data/stocks.csv")

chart = data * Mark(:line, point=true) * Encoding(
    "year(date):T",
    "mean(price):Q",
    color=:symbol,
)
